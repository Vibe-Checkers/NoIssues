FROM python:3.11-slim

LABEL maintainer="Python Requests Team" \
      description="Requests is a simple, yet elegant HTTP library for Python."

ENV PYTHONUNBUFFERED=1 \
    PYTHONDONTWRITEBYTECODE=1

WORKDIR /app

# System dependencies for building and SSL certificates
RUN apt-get update && \
    apt-get install -y --no-install-recommends \
        build-essential \
        ca-certificates \
        libssl-dev \
        libffi-dev \
        curl \
    && rm -rf /var/lib/apt/lists/*

# Copy packaging metadata first to leverage Docker layer caching
COPY pyproject.toml setup.py setup.cfg tox.ini ./

# Copy the rest of the source code
COPY . .

# Install Python build tools and the package itself
RUN python -m pip install --upgrade pip setuptools wheel && \
    python -m pip install --no-cache-dir .

# Create a non-root user for running the application
RUN useradd -m app && chown -R app /app
USER app

# Basic healthcheck to ensure the package is importable
HEALTHCHECK --interval=30s --timeout=5s CMD python -c "import requests; print(requests.__version__)" || exit 1

# Entry point: demonstrate a quick import and version print of the installed package
CMD ["python", "-c", "import requests; print('requests version:', requests.__version__)"]
